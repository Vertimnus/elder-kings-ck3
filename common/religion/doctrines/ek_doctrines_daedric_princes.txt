### These doctrines define your attitude towards the Daedric Princes
# Replaces the old 'Daedric Worship' criminal doctrine

# General view of the Daedra, there for history purposes (so you don't have to define every single Daedra if your faith hates/likes them allà
doctrine_hidden_daedra_blanket = {
	group = "special"
	
	is_available_on_create = { always = no }
	
	doctrine_hidden_daedra_allowed = {
		is_shown = { always = no }
	}
	
	doctrine_hidden_daedra_shunned = {
		is_shown = { always = no }
	}
	
	doctrine_hidden_daedra_criminal = {
		is_shown = { always = no }
	}
}

# Specific to each Daedra
doctrine_azura = {
	group = "daedric_princes"

	### Adds it to your faith's pantheon
	doctrine_azura_pantheon = {
		parameters = {
			adds_azura_pantheon_active = yes
			worship_azura_allowed_active = yes
		}
	}
	
	### Makes it just okay but not much more
	doctrine_azura_allowed = {
		parameters = {
			worship_azura_allowed_active = yes
		}
	}
	
	### Azura's worship is kinda yikes tbh
	doctrine_azura_shunned = {
		parameters = {
			worship_azura_shunned_active = yes
			hostility_override_doctrine_pantheon_azura = 2
		}
	}
	
	### They are CRIMINAL, and you hate them
	doctrine_azura_criminal = {
		parameters = {
			worship_azura_criminal_active = yes
			hostility_override_doctrine_pantheon_azura = 3
		}
	}
}

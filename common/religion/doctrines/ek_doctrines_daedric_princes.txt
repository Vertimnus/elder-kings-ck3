### These doctrines define your attitude towards the Daedric Princes
# Replaces the old 'Daedric Worship' criminal doctrine

# General view of the Daedra, there for history purposes (so you don't have to define every single Daedra if your faith hates/likes them all)
doctrine_hidden_daedra_blanket = {
	group = "special"
	
	is_available_on_create = { always = no }
	
	doctrine_hidden_daedra_allowed = {
		is_shown = { always = no }
	}
	
	doctrine_hidden_daedra_shunned = {
		is_shown = { always = no }
	}
	
	doctrine_hidden_daedra_criminal = {
		is_shown = { always = no }
	}
}

# Specific to each Daedra
doctrine_azura = {
	group = "daedric_princes"

	# Adds it to your faith's pantheon
	doctrine_azura_pantheon = {
		parameters = {
			adds_azura_pantheon_active = yes
			worship_azura_allowed_active = yes
			hostility_override_doctrine_azura_criminal = 3
		}
		
		can_pick = {
			custom_description = {
				text = faith_must_allow_daedric_patrons_trigger
				creating_faith_allow_daedric_patrons = yes
			}
			trigger_if = {
				limit = { NOT = { flag:doctrine_pantheon_daedra = { is_in_list = selected_doctrines } } }
				# If you use a pantheon other than 'the Daedric Princes' you need a reason to add that Daedra
				custom_description = {
					text = faith_must_allow_patron_azura_trigger
					creating_faith_doctrine_adds_azura = yes
				}
			}
		}
		
		piety_cost = {
			value = faith_doctrine_cost_massive

			# Multiplier for keeping same doctrine
			if = {
				limit = { has_doctrine = doctrine_azura_pantheon }
				multiply = faith_unchanged_doctrine_cost_mult
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_shunned }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_criminal }
				multiply = faith_changed_doctrine_cost_mult_three_step
			}
		}
	}
	
	# Makes it just okay but not much more
	doctrine_azura_allowed = {
		parameters = {
			worship_azura_allowed_active = yes
		}
		
		piety_cost = {
			value = faith_doctrine_cost_high

			# Multiplier for keeping same doctrine
			if = {
				limit = { has_doctrine = doctrine_azura_allowed }
				multiply = faith_unchanged_doctrine_cost_mult
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_criminal }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
		}
	}
	
	# Its worship is kinda yikes tbh
	doctrine_azura_shunned = {
		parameters = {
			worship_azura_shunned_active = yes
		}
		
		piety_cost = {
			value = faith_doctrine_cost_high

			# Multiplier for keeping same doctrine
			if = {
				limit = { has_doctrine = doctrine_azura_shunned }
				multiply = faith_unchanged_doctrine_cost_mult
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_pantheon }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
		}
	}
	
	# They are CRIMINAL, and you hate them
	doctrine_azura_criminal = {
		parameters = {
			worship_azura_criminal_active = yes
			hostility_override_doctrine_pantheon_azura = 3
			hostility_override_doctrine_azura_pantheon = 3
		}
		
		piety_cost = {
			value = faith_doctrine_cost_high

			# Multiplier for keeping same doctrine
			if = {
				limit = { has_doctrine = doctrine_azura_criminal }
				multiply = faith_unchanged_doctrine_cost_mult
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_allowed }
				multiply = faith_changed_doctrine_cost_mult_two_step
			}
			
			if = {
				limit = { has_doctrine = doctrine_azura_pantheon }
				multiply = faith_changed_doctrine_cost_mult_three_step
			}
		}
	}
}

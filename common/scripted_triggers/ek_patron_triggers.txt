is_worshipping_daedra = {
	OR = {
		has_doctrine_parameter = adds_$DAEDRA$_pantheon_active
		has_doctrine_parameter = worship_$DAEDRA$_allowed_active
	}
}

	# Doctrines and tenets are aligned with $DAEDRA$
	is_worshipping_daedra = { DAEDRA = $DAEDRA$ }
	# Faith is not centered around a single Prince, or if it is it's around this Prince
	has_doctrine_parameter = adds_$DAEDRA$_pantheon_active

# Daedra #
is_worshipping_daedra = {
	OR = {
		faith = {
			NOT = { has_doctrine = doctrine_daedra_worship_forbidden } #cant be criminal
			OR = {
                has_doctrine_parameter = allows_worship_$DAEDRA$_all_doctrines_active
                has_doctrine_parameter = allows_worship_$DAEDRA$_shunned_doctrines_active
                has_doctrine_parameter = allows_worship_$DAEDRA$_accepted_doctrines_active
            }
		}
	}
}

patron_daedra_valid = {
	# Doctrines and tenets are aligned with $DAEDRA$
	is_worshipping_daedra = { DAEDRA = $DAEDRA$ }
	# Faith is not centered around a single Prince
	OR = {
		faith = { has_doctrine_parameter = allows_worship_$DAEDRA$_only_active }
		NOT = { faith = { has_exclusive_daedric_prince_pantheon = yes } }
	}
}

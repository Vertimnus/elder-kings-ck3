# Daedra #
is_worshipping_daedra = {
	NOT = { has_doctrine_parameter = daedric_worship_criminal } #cant be criminal
	OR = {
		# Most Daedric faiths should have at least one tenet or doctrine bringing their Prince in the pantheon, but just in case
		has_doctrine_parameter = allows_worship_$DAEDRA$_only_active
		has_doctrine_parameter = allows_worship_$DAEDRA$_all_doctrines_active
		has_doctrine_parameter = allows_worship_$DAEDRA$_shunned_doctrines_active
		AND = {
			has_doctrine_parameter = allows_worship_$DAEDRA$_accepted_doctrines_active
			has_doctrine_parameter = daedric_worship_accepted
		}
	}
}

patron_daedra_valid = {
	# Doctrines and tenets are aligned with $DAEDRA$
	is_worshipping_daedra = { DAEDRA = $DAEDRA$ }
	# Faith is not centered around a single Prince, or if it is it's around this Prince
	OR = {
		has_doctrine_parameter = allows_worship_$DAEDRA$_only_active
		NOR	= {
			patron_exclusive_daedra = yes
			has_doctrine_parameter = pantheon_cannot_be_changed_active
		}
	}
}

# Daedra #
patron_azura_valid = {
	OR = {
		faith = {
			NOT = { has_doctrine = doctrine_daedra_worship_forbidden } #cant be criminal
			OR = {
				has_doctrine_parameter = allows_worship_azura_all_doctrines_active
				has_doctrine_parameter = allows_worship_azura_shunned_doctrines_active
				has_doctrine_parameter = allows_worship_azura_accepted_doctrines_active
			}
		}
	}
}

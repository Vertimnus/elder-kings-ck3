has_aptitude = {
	custom_description = {
		text = aptitude_greater_or_equal
		value = $VALUE$
		aptitude >= $VALUE$
	}
}

has_aptitude_less_than = {
	custom_description = {
		text = aptitude_less_than
		value = $VALUE$
		aptitude < $VALUE$
	}
}

has_aptitude = {
	custom_description = {
		text = aptitude_greater_or_equal
		value = $VALUE$
		aptitude >= $VALUE$
	}
}

has_aptitude_less_than = {
	custom_description = {
		text = aptitude_less_than
		value = $VALUE$
		aptitude < $VALUE$
	}
}

has_magicka = {
	custom_description = {
		text = magicka_greater_or_equal
		value = $VALUE$
		trigger_if = {
			limit = { has_variable = magicka }
			var:magicka >= $VALUE$
		}
		trigger_else = { always = no }
	}
}

has_magicka_less_than = {
	custom_description = {
		text = magicka_less_than
		value = $VALUE$
		trigger_if = {
			limit = { has_variable = magicka }
			var:magicka < $VALUE$
		}
		trigger_else = { always = yes }
	}
}

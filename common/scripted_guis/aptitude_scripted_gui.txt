magicka_casting_spell = { #observer in hud.gui actually triggers spellbook widget
    scope = character
    is_shown = { has_variable = spell_recipient }
    effect = {
        remove_variable = spell_selected
        remove_variable = spell_recipient
    }
}

spell_flames_caster = {
    scope = character

    is_shown = {
        #spell_knows_fireball = yes
        always = yes #for testing
    }

    is_valid = {
        has_magicka = { VALUE = 5 }
        #always = yes #for testing
    }

    effect = {
        hidden_effect = {
            set_variable = { name = spell_selected value = flag:flames }
        }
        custom_label = spell_name_caster
        show_as_tooltip = {
            add_magicka = { VALUE = -10 }
        }
    }
}

spell_flames_recipient = {
    scope = character

    is_valid = {
        var:spell_selected = flag:flames
    }

    effect = {
        custom_label = spell_name_castee
        show_as_tooltip = {
            add_aptitude_skill = { VALUE = 1 }
        }
    }
}

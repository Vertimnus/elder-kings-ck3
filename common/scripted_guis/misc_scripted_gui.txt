culture_group_cultures = {
    scope = culture_group

    effect = {
        every_in_global_list = { #TODO replace w/ every/ordered_culture_global when 1.3.x
            variable = culture_list
            limit = { ROOT = THIS.culture_group }
            if = {
                #limit = { any_county = { culture = PREV.PREV } }
                limit = {
                    is_target_in_global_variable_list = {
                        name = culture_is_alive
                        target = this
                    }
                }
                custom_tooltip = CULTURE_NAME
            }
            else = { custom_tooltip = CULTURE_NAME_DEAD }
        }
    }
}

### These titles are based on regions, which are tied to nativity regions
# Only cultures natives from the region the kingdoms are based on should be able to unite them
# Iron Orc can unite Craglorn (since they come from here), but Eastmarchers can't
form_craglorn_kingdom_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"

	desc = form_craglorn_kingdom_decision_desc
	selection_tooltip = form_craglorn_kingdom_decision_tooltip
	major = yes

	ai_check_interval = 120

	is_shown = {
		is_landed = yes
		exists = capital_province
		
		can_unify_kingdom_of = { REGION = craglorn }
		
		unify_kingdom_government_trigger = yes
	}

	is_valid = {
		completely_controls_region = mundus_tamriel_hammerfell_craglorn
		unify_kingdom_law_trigger = yes
	}

	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_at_war = no
	}

	cost = {
		gold = 500
	}

	effect = {
		form_craglorn_scripted_effect = yes
		add_to_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:flag_formed_craglorn_kingdom
		}
		
		trigger_event = { id = ektitulartitle.0001 }
	}
	
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}

### Unite the Reach
form_reach_empire_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"

	desc = form_reach_empire_decision_desc
	selection_tooltip = form_reach_empire_decision_tooltip
	major = yes

	ai_check_interval = 120

	is_shown = {
		is_landed = yes
		highest_held_title_tier <= tier_kingdom
		exists = capital_province
		unify_kingdom_government_trigger = yes
		
		OR = {
			AND = {
				is_native_to = { REGION = markarth CULTURE = root.culture }
				capital_province = { geographical_region = custom_nativity_markarth }
			}
			AND = {
				is_native_to = { REGION = craglorn CULTURE = root.culture }
				capital_province = { geographical_region = custom_nativity_craglorn }
			}
			AND = {
				is_native_to = { REGION = greater_wrothgar CULTURE = root.culture }
				capital_province = { geographical_region = custom_nativity_greater_wrothgar }
			}
		}
		
		NOT = {  #Can only do it once.
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:flag_created_reach_empire
			}
		}
	}

	is_valid = {
		completely_controls_region = mundus_tamriel_reach_east_reach
		completely_controls_region = mundus_tamriel_reach_west_reach
		completely_controls_region = mundus_tamriel_reach_north_reach
		completely_controls_region = mundus_tamriel_reach_south_reach
		unify_kingdom_law_trigger = yes
		highest_held_title_tier = tier_kingdom
		prestige_level >= 3
	}

	is_valid_showing_failures_only = {
		is_available_adult = yes
	}

	cost = {
		# Based on Revive Greater Armenia
		prestige = 2000
		piety = 1000
	}

	effect = {
		show_as_tooltip = {
			form_reach_empire_scripted_effect = yes
		}
		save_scope_as = founder
		trigger_event = {
			id = ektitulartitle.1001
		}
		#Notify other players.
		every_player = {
			limit = {
				### TODO: Uncomment it
				# NOT = { this = root }
				is_within_diplo_range = { CHARACTER = root }
			}
			trigger_event = ektitulartitle.1002
		}
		hidden_effect = {
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:flag_created_reach_empire
			}
		}
	}
	
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}

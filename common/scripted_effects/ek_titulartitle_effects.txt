form_craglorn_scripted_effect = {
	save_scope_as = scoped_ruler
	title:k_craglorn = { save_scope_as = craglorn }
	#De jure shifts.
	#K_Elinhir
	title:d_belkarth = { set_de_jure_liege_title = title:k_craglorn }
	title:d_elinhir = { set_de_jure_liege_title = title:k_craglorn }
	title:d_spellscar = { set_de_jure_liege_title = title:k_craglorn }
	#K_Dragonstar
	title:d_abdir = { set_de_jure_liege_title = title:k_craglorn }
	title:d_dragonstar = { set_de_jure_liege_title = title:k_craglorn }
	title:d_hel_ra = { set_de_jure_liege_title = title:k_craglorn }
	title:d_lankara = { set_de_jure_liege_title = title:k_craglorn }
	title:d_skyreach = { set_de_jure_liege_title = title:k_craglorn }
	
	#make sure to set hammerfell de jure
	title:k_craglorn = { set_de_jure_liege_title = title:e_hammerfell}


	else = { custom_tooltip = form_craglorn_kingdom_decision_effect_tooltip }
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change
		add_claim_on_loss = no
	}
	title:k_craglorn = {
		change_title_holder = {
			holder = root
			change = scope:change
		}
	}
	resolve_title_and_vassal_change = scope:change
	hidden_effect = { 
		set_primary_title_to = title:k_craglorn 
		destroy_title = title:k_elinhir
		destroy_title = title:k_dragonstar
	}
	add_prestige = major_prestige_gain
}

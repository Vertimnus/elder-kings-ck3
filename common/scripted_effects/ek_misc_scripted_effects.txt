save_culture_in_global_list = {
	$CULTURE$ = { save_scope_as = culture_to_save }
	add_to_global_variable_list = {
		name = culture_list
		target = scope:culture_to_save
	}
}

set_culture_native_to = {
	add_to_global_variable_list = {
		name = culture_innovation_native_$REGION$
		target = $CULTURE$
	}
	$CULTURE$ = { add_innovation = innovation_native_$REGION$ }
}

# Effect run when attempting to become an Occultist as a Witch
witch_become_occultist_effect = {
	remove_trait = witch
	add_trait = occultist
	
	add_piety_level = 1
}

startup_culture_effect = {
	save_culture_in_global_list = { CULTURE = culture:$CULTURE$ }
	culture_key_setup = { CULTURE = $CULTURE$ }
}

culture_key_setup = {
    culture:$CULTURE$ = {
        set_variable = { name = culture_key value = flag:$CULTURE$_ }
    }
}

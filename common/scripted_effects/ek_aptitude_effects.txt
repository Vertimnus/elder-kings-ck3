# add_aptitude_skill = { VALUE = x SUBTRACT = no/yes}
add_aptitude_skill = {
    if = {
        limit = { NOT = { exists = var:aptitude_base } }
        set_variable = { name = aptitude_base value = 0 }
    }
    custom_description = {
        text = add_aptitude_skill
        value = $VALUE$
        change_variable = { name = aptitude_base add = $VALUE$ }
    }
}
